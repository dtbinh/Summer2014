1function xkPlus = update_eq_unknown_state(xk,vk,uk)
 2
 3global Ts;
 4
 5%xk: 17+5*Nf x 2N+1, inertial position, inertial velocity, inertial quaternion, relative position, and relative quaternion ... then unknown feature parameters 
 6%vk: 12 x 2N+1, noise on wi,(wj_est),ai,(vj_inertial)
 7
 8%uk[wi;ai;mag_i] : 9 x 2N+1
 9
10n = size(xk,1);
11v = size(vk,1);
12
13% IMU measurements of me
14wi_m = uk(1:3);
15ai_m = uk(4:6);
16
17xkPlus = zeros(size(xk));
18for k = 1:size(xk,2);
19    xhat = xk(:,k);
20    % hist position relative to my. My body frame
21    rji = xhat(11:13);
22    % my velocity
23    vin = xhat(4:6);
24    % my attitude
25    qin = xhat(7:10);
26    % his attitude relative to mine. His body frame
27    qji = xhat(14:17);
28    % the unknown feature locations are in the inertial frame and DO NOT update 
29    
30    vn = vk(:,k);
31    xdot = zeros(n,1);
32    
33    vj = vn(10:12);
34    wj = vn(4:6);
35    
36    wi = wi_m + vn(1:3);
37    ai = ai_m + vn(7:9);
38    
39    % cosine matrix
40    Cin = attparsilentmex(qin,[6 1]);
41    % time rate of my position
42    xdot(1:3) = Cin'*vin;
43    % time rate of my velocity
44    xdot(4:6) = ai - squiggle(wi)*vin;
45    % time rate of my attitude
46    % A matrix for qin
47    Ain = 0.5*[ -qin(2:4)';qin(1)*eye(3) + squiggle(qin(2:4))];
48    xdot(7:10) = Ain*wi;
49
50    % relative attitude cosine matrix
51    Cji = attparsilentmex(qji,[6 1]);
52
53    % relative angular velocity in j frame
54    wji = wj-Cji*wi;
55    % A matrix for q_ji
56    A = 0.5*[ -qji(2:4)';qji(1)*eye(3) + squiggle(qji(2:4))];
57    % time rate of relative attitude
58    xdot(14:17) = A*wji;
59    
60    % time rate of relative position
61    xdot(11:13) = (vj-vin) - squiggle(wi)*rji;
62
63    xkPlus(:,k) = xhat + Ts*xdot;
64    %re-normalize quaternions
65    xkPlus(7:10,k) = xkPlus(7:10,k)/norm(xkPlus(7:10,k));
66    xkPlus(14:17,k) = xkPlus(14:17,k)/norm(xkPlus(14:17,k));
67end
68
69end
