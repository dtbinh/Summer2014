1function [xnew,Pnew] = ukf_update_10_state(xk,Pxk,Pvk,Pnk,uk,ytilde,alpha)
  2% Pxk: covariance associated with state
  3% Pvk: covariance associated with process noise
  4% Pnk: covariance assocaited with measurement noise
  5
  6% vector lengths
  7n = length(xk);
  8m = length(uk);
  9nl = size(Pnk,1);
 10vl = size(Pvk,1);
 11N = n+nl+vl;
 12
 13% number of known features seen by agents i and j (1 and 2)
 14m1 = uk(7);
 15
 16% constant gamma that influences sigma points
 17if nargin < 7
 18    alpha = 1e-3;
 19end
 20Kappa = 0;
 21lambda = alpha^2*(N+Kappa)-N;
 22gamm = sqrt(N+lambda);
 23
 24%augmented covariance and state
 25Paug = zeros(N);
 26Paug(1:n,1:n) = Pxk;
 27Paug(n+(1:vl),n+(1:vl)) = Pvk;
 28Paug(n+vl+(1:nl),n+vl+(1:nl)) = Pnk;
 29
 30xaug = [xk;zeros(nl+vl,1)];
 31
 32[Psq,resnorm] = sqrtm(Paug);
 33if (abs(resnorm) > 1e-14)
 34    fprintf('Error: matrix square root residual norm is %g.\n',resnorm);
 35    xnew = 0;
 36    Pnew = 0;
 37    return;
 38end
 39Psq = real(Psq);
 40
 41% compute the sigma points
 42XAUG = zeros(N,2*N+1);
 43XAUG(:,1) = xaug;
 44for k = 1:N
 45    XAUG(:,k+1) = xaug + gamm*Psq(:,k);
 46    XAUG(:,N+1+k) = xaug - gamm*Psq(:,k);
 47end
 48
 49% propagate the sigma points
 50XAUGPLUS = update_eq_10_state(XAUG(1:n,:),XAUG(n+(1:vl),:),uk);
 51
 52% weights
 53beta = 2;% optimal for Gaussian
 54wc = 0.5/(N+lambda)*ones(2*N+1,1);
 55wm = 0.5/(N+lambda)*ones(2*N+1,1);
 56wm(1) = lambda/(N+lambda);
 57wc(1) = lambda/(N+lambda) + 1-alpha^2+beta;
 58
 59% calculate the update
 60xp = sum(XAUGPLUS(1:n,:).*repmat(wm',n,1),2);
 61
 62% re-normalize
 63xp(7:10) = xp(7:10)/norm(xp(7:10));
 64
 65Pp = zeros(n);
 66for k = 1:2*N+1
 67    Pp = Pp + wc(k)*(XAUGPLUS(1:n,k)-xp)*(XAUGPLUS(1:n,k)-xp)';
 68end
 69
 70% pass the sigma points through the measurement function
 71YKAUG = measurement_eq_10_state(XAUGPLUS(1:n,:),XAUG(n+vl+(1:nl),:),uk);
 72
 73g = size(YKAUG,1);
 74
 75yhat = sum(YKAUG.*repmat(wm',g,1),2);
 76% add something to handle magnetometer normalization if necessary here
 77
 78% measurement covariance
 79Pyk = zeros(g);
 80% cross covariance
 81Pxkyk = zeros(n,g);
 82for k = 1:2*N+1
 83    Pyk = Pyk + wc(k)*(YKAUG(:,k)-yhat)*(YKAUG(:,k)-yhat)';
 84    Pxkyk = Pxkyk + wc(k)*(XAUGPLUS(1:n,k)-xp)*(YKAUG(:,k)-yhat)';
 85end
 86
 87% Kalman gain
 88if any(any(isnan(Pyk)))
 89    disp('Singular measurement covariance');
 90    xnew = 0;
 91    Pnew = 0;
 92    return;
 93end
 94
 95if rank(Pyk) < length(Pyk)
 96    disp('pause');
 97end
 98
 99Kk = Pxkyk*(Pyk\eye(g));
100
101% minimize diff between bearing/declination to features
102m1 = uk(7);
103yhat( (2:3:(3*m1))) = minangle(yhat( (2:3:(3*m1))),ytilde( (2:3:(3*m1))));
104yhat( (3:3:(3*m1))) = minangle(yhat( (3:3:(3*m1))),ytilde( (3:3:(3*m1))));
105xn = xp + Kk*(ytilde-yhat);
106
107% renormalize quaternions as necessary
108xn(7:10) = xn(7:10)/norm(xn(7:10));
109xnew = xn;
110Pnew = Pp - Kk*Pyk*Kk';
111end
