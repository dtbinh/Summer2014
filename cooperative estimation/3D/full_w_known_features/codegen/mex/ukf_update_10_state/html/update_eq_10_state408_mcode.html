1function xkPlus = update_eq_10_state(xk,vk,uk)
 2
 3global Ts;
 4
 5%xk: 10 x 2N+1, inertial position, inertial velocity, inertial quaternion, relative position, and relative quaternion
 6%vk: 6 x 2N+1, noise on wi,ai
 7
 8%uk[wi;ai] : 6 x 2N+1
 9
10n = size(xk,1);
11v = size(vk,1);
12
13% IMU measurements of me
14wi_m = uk(1:3);
15ai_m = uk(4:6);
16
17xkPlus = zeros(size(xk));
18for k = 1:size(xk,2);
19    xhat = xk(:,k);
20    % my velocity
21    vin = xhat(4:6);
22    % my attitude
23    qin = xhat(7:10);
24    
25    vn = vk(:,k);
26    xdot = zeros(10,1);
27    
28    wi = wi_m + vn(1:3);
29    ai = ai_m + vn(4:6);
30    
31    % cosine matrix
32    Cin = attparsilentmex(qin,[6 1]);
33    % time rate of my position
34    xdot(1:3) = Cin'*vin;
35    % time rate of my velocity
36    xdot(4:6) = ai - squiggle(wi)*vin;
37    % time rate of my attitude
38    % A matrix for qin
39    Ain = 0.5*[ -qin(2:4)';qin(1)*eye(3) + squiggle(qin(2:4))];
40    xdot(7:10) = Ain*wi;
41
42    xkPlus(:,k) = xhat(:) + Ts*xdot;
43    %re-normalize
44    xkPlus(7:10,k) = xkPlus(7:10,k)/norm(xkPlus(7:10,k));
45end
46
47end
