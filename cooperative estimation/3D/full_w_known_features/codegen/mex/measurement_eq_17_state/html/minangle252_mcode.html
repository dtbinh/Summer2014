1function [x1] = minangle(x1,x2)
 2%find the minimum angle between two vectors of 2D angles x1 and x2
 3count = 0;
 4while any( (abs(x1-2*pi-x2) < abs(x1-x2)) | (abs(x1+2*pi-x2) < abs(x1-x2)) )
 5    in1 = find( abs(x1-2*pi-x2) < abs(x1-x2) );
 6    x1(in1) = x1(in1)-2*pi;
 7    in2 = find( abs(x1+2*pi-x2) < abs(x1-x2) );
 8    x1(in2) = x1(in2)+2*pi;
 9    count = count+1;
10    if count > 50
11        disp('Warning: minangle() failing to converge');
12        return;
13    end
14end
15end
