1function [Y] = attparsilentmex(X,I,options)
  2%Tim Woodbury
  3%modified for AERO 622
  4%
  5%[Y] = attpar(X,I,options)
  6%Function to convert between attitude parametrizations
  7%
  8%INPUTS:
  9%X - matrix input of appropriate dimension (detailed later)
 10%I - 2 x 1 indexing vector indicating the input (I(1)) and output (I(2))
 11%    attitude parametrizations, corresponding to the numbers in the section,
 12%    "I/O SPECIFICATION PARAMETERS".
 13%options - a data structure. Currently only allows output Euler angle
 14%    sequence to be defined. Supported members are "seq" which should be a
 15%    [3 x 1] vector describing the first, second, and third rotations in
 16%    the desired output sequence.
 17%
 18%OUTPUTS:
 19%Y - output matrix of appropriate dimensions.
 20%    All angles are in radians.
 21%
 22%I/O SPECIFICATION PARAMETERS:
 23%1 - Direction cosine matrix, dimensions [3 x 3]
 24%2 - Euler principal axis/angle, [3 x 2]. [:,1] is the principal axis;
 25%    [1,2] is the principal angle (rad).
 26%3 - 2-angle parametrization, [3 x 4]
 27%4 - Euler angle sequence, [3 x 2]. [:,1] are the rotation angles in radians, 
 28%     and [2,1]-[2,2]-[2,3] is the rotation sequence. 
 29%     [3-1-3] sequence is default for output. Other sequences may be specified
 30%     by passing an optional data structure with a [3 x 1] member "seq" whose
 31%     entries [a;b;c] correspond to the sequence a-b-c. Any input sequence
 32%     may be used by specifying the second column of input appropriately.
 33%5 - Classical Rodrigues parameters [3 x 1]
 34%6 - quaternion [4 x 1]. The scalar part of the quaternion comes FIRST.
 35%7 - modified Rodrigues parameters [3 x 1]
 36%8 - exponential matrix, [3 x 3]
 37%9 - Cayley-Klein parameters, [2 x 2]
 38
 39%check if any options are passed
 40if nargin >=3
 41    EULOUTSEQ = options.seq;
 42    if isempty(EULOUTSEQ)
 43        EULOUTSEQ = [3;1;3];
 44    end
 45end
 46
 47if mod(I(1),floor(I(1)))
 48    error('Specified input parametrization must be of integer type.');
 49end
 50if mod(I(2),floor(I(2)))
 51    error('Specified input parametrization must be of integer type.');
 52end
 53
 54inp = I(1);
 55outp = I(2);
 56
 57%%check if output and inp are the same - if so do nothing
 58if I(1) == I(2)
 59    %disp('Warning - input and output parametrizations are the same.');
 60    Y = X;
 61else
 62    %for each inp, convert to DCM
 63    switch inp
 64        case 1 %DCM
 65            %disp('Input value specified as direction cosine matrix.');
 66            if size(X)~=[3,3]
 67                error('Input dimensions invalid for DCM.');
 68            end
 69            DCM = X;
 70            %check orthogonality of DCM using Frobenius norm:
 71            Cinv = inv(DCM);
 72            Ctrans = DCM';
 73            err = norm(Cinv-Ctrans,'fro');
 74            if err>=1e-4
 75                %disp('Matrix input is not orthonormal to 4 decimal places, measured by the Frobenius norm.');
 76                %disp(['Error is ' num2str(err) '.']);
 77                disp('Terminating after nonorthogonal matrix specification.');
 78            elseif err>1e-15
 79                %disp('Warning: cosine matrix input is not orthogonal to machine precision.');
 80                %disp(['Frobenius norm satisfies orthogonality to ' num2str(-floor(log10(err))) ' decimal places.']);
 81                %disp('Calclation proceeding:');
 82            end
 83        case 2 %principal axis/angle
 84            %disp('Input value specified as Euler principal axis/angle.');
 85            if size(X)~=[3,2]
 86                error('Input dimensions invalid for principal axis/angle.');
 87            end
 88            PHI_in = X(1,2);
 89            e_in = X(:,1);
 90            DCM = eye(3)*cos(PHI_in)+(1-cos(PHI_in))*(e_in*e_in')-squiggle(e_in)*sin(PHI_in);
 91        case 3 %2-angle parametrization
 92            %X: [3 x 4]
 93            %extract e_a, PHI_a, e_b, PHI_b from X:
 94            %disp('Input value specified as two-angle parametrization.');
 95            if size(X)~=[3,4]
 96                error('Input dimensions invalid for two-angle parametrization.');
 97            end
 98            e_a = X(:,1);
 99            e_b = X(:,3);
100            PHI_a = X(1,2);
101            PHI_b = X(1,4);
102            %compute cosine matrices C_a and C_b
103            C_a = eye(3)*cos(PHI_a)+(1-cos(PHI_a))*(e_a*e_a')-squiggle(e_a)*sin(PHI_a);
104            C_b = eye(3)*cos(PHI_b)+(1-cos(PHI_b))*(e_b*e_b')-squiggle(e_b)*sin(PHI_b);
105            DCM = C_b*C_a;
106        case 4 %Euler angles
107            %disp(['Input value specified as Euler angle sequence ' num2str(X(1,2)) '-' num2str(X(2,2)) '-' num2str(X(3,2)) '.']);
108            if (size(X)~=[3,2])
109                if size(X)==[3,1]
110                    error('Rotation sequence not specified.');
111                else
112                    error('Input dimensions invalid for Euler angle sequence.');
113                end
114            end
115            %extract rotation sequence
116            r1 = X(1,2);
117            r2 = X(2,2);
118            r3 = X(3,2);
119            if (r2==r1)||(r2==r3)
120                error('Specified second rotation axis must be different from 1st and 3rd rotation axis.')
121            end
122            %extract rotation angles
123            phi1 = X(1,1);
124            phi2 = X(2,1);
125            phi3 = X(3,1);
126            %for symmetric, second angle should be POSITIVE. If not, may cause
127            %problems. Autoconvert but display warning.
128            if phi2<0
129                phi2 = mod(phi2,pi*2);
130                %disp('Warning - second Euler angle specified is less than 0 radians, may cause problems!');
131                %disp('Attempting to resolve by converting to positive angle via modulus operation.');
132            end
133            %determine rotation matrices
134            R1 = DCMConverter(r1,phi1);
135            R2 = DCMConverter(r2,phi2);
136            R3 = DCMConverter(r3,phi3);
137            DCM = R3*R2*R1;
138        case 5 %Rodrigues parameters
139            %disp('Input value specified as classic Rodrigues parameters.');
140            if size(X)~=[3,1]
141                error('Input dimensions invalid for Rodrigues parameters.');
142            end
143            rho = X;
144           %DCM = 1/(1+rho'*rho)*((1-rho'*rho)*eye(3)+2*(rho*rho')-2*squiggle(rho));
145           DCM = eye(3)+2./(1+rho'*rho)*(-squiggle(rho)+squiggle(rho)*squiggle(rho));
146        case 6 %quaternion
147            %disp('Input value specified as quaternion.');
148            if length(X)~=4
149                error('Input dimensions invalid for quaternion.');
150            end
151            q1 = X(2);
152            q2 = X(3);
153            q3 = X(4);
154            q4 = X(1);
155            DCM = [q1^2-q2^2-q3^2+q4^2 2*(q1*q2+q3*q4) 2*(q1*q3-q2*q4);
156                2*(q1*q2-q3*q4) q2^2-q1^2-q3^2+q4^2 2*(q2*q3+q1*q4);
157                2*(q1*q3+q2*q4) 2*(q2*q3-q1*q4) q3^2-q1^2-q2^2+q4^2];
158        case 7 %modified Rodrigues parameters
159            %disp('Input value specified as modified Rodrigues parameters.');
160            if size(X)~=[3,1]
161                error('Input dimensions invalid for modified Rodrigues parameters.');
162            end
163            sigma = X;
164            %DCM = eye(3)+(8*squiggle(sigma)*squiggle(sigma)-4*(1-(sigma'*sigma))*squiggle(sigma))./((1+(sigma'*sigma))^2);
165            DCM = eye(3)-4*(1-sigma'*sigma)/(1+sigma'*sigma)^2*squiggle(sigma)+8/(1+sigma'*sigma)^2*squiggle(sigma)*squiggle(sigma);
166        otherwise
167            error('Invalid parametrization specified.');
168    end
169    %convert DCM to output form
170    switch outp
171        case 1%DCM
172            %disp('Output value specified as direction cosine matrix.');
173            Y = DCM;
174        case 2%principal axis/angle
175            %disp('Output value specified as Euler principal axis/angle.');
176            ksi = trace(DCM);
177            PHI = acos((ksi-1)/2);
178            e = zeros(3,1);
179            if ((round(ksi*1e4)/1e4)==-1)
180                e(1) = sqrt((1+DCM(1,1))*.5);
181                e(2) = sqrt((1+DCM(2,2))*.5);
182                e(3) = sqrt((1+DCM(3,3))*.5);
183                %create a vertical array of the sign of the products [c3c1;c2c3;c1c2]; 
184                signs = [sign(DCM(3,1)); sign(DCM(2,3)); sign(DCM(1,2))];
185                %set the negative values to zero for a binary representation
186                signs(find(signs==-1))=0;
187                % compute binary from signs
188                signs = signs(3)*4 + signs(2)*2 + signs(1)*1;
189                %depending on the result, change signs.
190                switch signs
191                    case 1
192                        e(3) = -e(3);
193                    case 2
194                        e(2) = -e(2);
195                        e(3) = -e(3);
196                    case 4
197                        e(2) = -e(2);
198                    otherwise
199                        %do nothing
200                end
201            else
202                e(1) = (DCM(2,3)-DCM(3,2));
203                e(2) = (DCM(3,1)-DCM(1,3));
204                e(3) = (DCM(1,2)-DCM(2,1));
205                e=e./sqrt((1+ksi)*(3-ksi));
206            end
207            %e(1) = (DCM(2,3)-DCM(3,2))/(2*sin(PHI));
208            %e(2) = (DCM(3,1)-DCM(1,3))/(2*sin(PHI));
209            %e(3) = (DCM(1,2)-DCM(2,1))/(2*sin(PHI));
210            
211            Y = zeros(3,2);
212            Y(:,1) = e;
213            Y(1,2) = PHI;
214        case 3%2-angle parametrization
215            %disp('Output value specified as two-angle parametrization.');
216            C_a = [cos(pi/4) sin(pi/4) 0;
217                -sin(pi/4) cos(pi/4) 0;
218                0 0 1];
219            C_b = DCM*C_a';
220            %decompose C_a into e_a, PHI_a
221            PHI_a = acos((trace(C_a)-1)/2);
222            e_a = zeros(3,1);
223            e_a(1) = (C_a(2,3)-C_a(3,2))/(2*sin(PHI_a));
224            e_a(2) = (C_a(3,1)-C_a(1,3))/(2*sin(PHI_a));
225            e_a(3) = (C_a(1,2)-C_a(2,1))/(2*sin(PHI_a));
226            %decompose C_b into e_b, PHI_b
227            PHI_b = acos((trace(C_b)-1)/2);
228            e_b = zeros(3,1);
229            e_b(1) = (C_b(2,3)-C_b(3,2))/(2*sin(PHI_b));
230            e_b(2) = (C_b(3,1)-C_b(1,3))/(2*sin(PHI_b));
231            e_b(3) = (C_b(1,2)-C_b(2,1))/(2*sin(PHI_b));
232            %assign output
233            Y = zeros(3,4);
234            Y(:,1) = e_a;
235            Y(1,2) = PHI_a;
236            Y(:,3) = e_b;
237            Y(1,4) = PHI_b;
238        case 4%Euler angles
239            %use a 3-1-3 rotation sequence as a default;
240            %if ~exist('EULOUTSEQ','var')
241            %    EULOUTSEQ = [3;1;3];
242            %end
243            %disp(['Output value specified as Euler angle sequence ' num2str(EULOUTSEQ(1)) '-' num2str(EULOUTSEQ(2)) '-' num2str(EULOUTSEQ(3)) '.']);
244            [eul, seq] = DCM2eul(DCM,EULOUTSEQ);
245            %verify sequence by computing error norm of reconstruction
246            froErr = norm(DCMConverter(seq(3),eul(3))*DCMConverter(seq(2),eul(2))*DCMConverter(seq(1),eul(1))-DCM,'fro');
247            if froErr>1e-6
248                %disp(['Warning: Frobenius norm of error in DCM reconstruction from Euler angles is ' num2str(froErr) '.']);
249            end
250            Y = [eul seq];
251        case 5%Rogrigues parameters
252            %disp('Output value specified as classic Rodrigues parameters.');
253            ksi = trace(DCM);
254            rho = zeros(3,1);
255            rho(1)=DCM(2,3)-DCM(3,2);
256            rho(2)=DCM(3,1)-DCM(1,3);
257            rho(3)=DCM(1,2)-DCM(2,1);
258            rho = rho./(1+ksi);
259            Y = rho;
260        case 6%quaternion
261            %disp('Output value specified as quaternion.');
262            q1 = 1/2*sqrt(1+DCM(1,1)-DCM(2,2)-DCM(3,3));
263            q2 = 1/2*sqrt(1-DCM(1,1)+DCM(2,2)-DCM(3,3));
264            q3 = 1/2*sqrt(1-DCM(1,1)-DCM(2,2)+DCM(3,3));
265            q4 = 1/2*sqrt(1+DCM(1,1)+DCM(2,2)+DCM(3,3));%scalar part
266            %determine the maximum of the four and use this value to determine the
267            %other parameters
268            [~,iMax] = max([q1;q2;q3;q4]);
269            switch iMax
270                case 1
271                    qV = 1/(4*q1)*[DCM(1,2)+DCM(2,1);DCM(1,3)+DCM(3,1);DCM(2,3)-DCM(3,2)];
272                    q = [q1;qV];
273                case 2
274                    qV = .25/q2*[DCM(2,1)+DCM(1,2);DCM(2,3)+DCM(3,2);DCM(3,1)-DCM(1,3)];
275                    q = [qV(1);q2;qV(2:3)];
276                case 3
277                    qV = .25/q3*[DCM(3,1)+DCM(1,3);DCM(3,2)+DCM(2,3);DCM(1,2)-DCM(2,1)];
278                    q = [qV(1:2);q3;qV(3)];
279                case 4
280                    qV = .25/q4*[DCM(2,3)-DCM(3,2);DCM(3,1)-DCM(1,3);DCM(1,2)-DCM(2,1)];
281                    q = [qV;q4];
282            end
283            %output in the format [qScalar;qVector]
284            Y = [q(4);q(1);q(2);q(3)];
285        case 7 %modified Rodrigues parameters
286            %disp('Output value specified as modified Rodrigues parameters.');
287            ksi = trace(DCM);
288            s = zeros(3,1);
289            %if trace(C) is within 1e-4 of -1, then assume ksi = -1:
290            if (round(ksi*1e4)/1e4)==-1
291                s(1) = sqrt((1+DCM(1,1))*.5);
292                s(2) = sqrt((1+DCM(2,2))*.5);
293                s(3) = sqrt((1+DCM(3,3))*.5);
294                %create a vertical array of the sign of the products [c3c1;c2c3;c1c2]; 
295                signs = [sign(DCM(3,1)); sign(DCM(2,3)); sign(DCM(1,2))];
296                %set the negative values to zero for a binary representation
297                signs(find(signs==-1))=0;
298                % compute binary from signs
299                signs = signs(3)*4 + signs(2)*2 + signs(1)*1;
300                %depending on the result, change signs.
301                switch signs
302                    case 1
303                        s(3) = -s(3);
304                    case 2
305                        s(2) = -s(2);
306                        s(3) = -s(3);
307                    case 4
308                        s(2) = -s(2);
309                    otherwise
310                        %do nothing
311                end
312            %else, perform the usual computation:
313            else
314                k = 1/((sqrt(ksi+1)+2)*(sqrt(ksi+1)));
315                s(2) = k*(DCM(3,1)-DCM(1,3));
316                s(3) = k*(DCM(1,2)-DCM(2,1));
317                s(1) = k*(DCM(2,3)-DCM(3,2));
318            end
319            %determine principal axis
320%             PHI = acos((trace(DCM)-1)/2);
321%             e = zeros(3,1);
322%             e(1) = (DCM(2,3)-DCM(3,2))/(2*sin(PHI));
323%             e(2) = (DCM(3,1)-DCM(1,3))/(2*sin(PHI));
324%             e(3) = (DCM(1,2)-DCM(2,1))/(2*sin(PHI));
325%             sigma = e*tan(PHI/4);
326            Y = s;
327        otherwise
328            error('Specified output parametrization not available.');
329    end
330end
331end
