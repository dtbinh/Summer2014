1function xkPlus = update_eq_17_state(xk,vk,uk)
 2
 3global Ts;
 4
 5%xk: 17 x 2N+1, inertial position, inertial velocity, inertial quaternion, relative position, and relative quaternion
 6%vk: 12 x 2N+1, noise on wi,(wj_est),ai,(vj_inertial)
 7
 8%uk[wi;ai;mag_i] : 9 x 2N+1
 9
10n = size(xk,1);
11v = size(vk,1);
12
13% IMU measurements of me
14wi_m = uk(1:3);
15ai_m = uk(4:6);
16
17xkPlus = zeros(size(xk));
18for k = 1:size(xk,2);
19    xhat = xk(:,k);
20    % hist position relative to my. My body frame
21    rji = xhat(11:13);
22    % my velocity
23    vin = xhat(4:6);
24    % my attitude
25    qin = xhat(7:10);
26    % his attitude relative to mine. His body frame
27    qji = xhat(14:17);
28    
29    vn = vk(:,k);
30    xdot = zeros(17,1);
31    
32    vj = vn(10:12);
33    wj = vn(4:6);
34    
35    wi = wi_m + vn(1:3);
36    ai = ai_m + vn(7:9);
37    
38    % cosine matrix
39    Cin = attparsilentmex(qin,[6 1]);
40    % time rate of my position
41    xdot(1:3) = Cin'*vin;
42    % time rate of my velocity
43    xdot(4:6) = ai - squiggle(wi)*vin;
44    % time rate of my attitude
45    % A matrix for qin
46    Ain = 0.5*[ -qin(2:4)';qin(1)*eye(3) + squiggle(qin(2:4))];
47    xdot(7:10) = Ain*wi;
48
49    % relative attitude cosine matrix
50    Cji = attparsilentmex(qji,[6 1]);
51
52    % relative angular velocity in j frame
53    wji = wj-Cji*wi;
54    % A matrix for q_ji
55    A = 0.5*[ -qji(2:4)';qji(1)*eye(3) + squiggle(qji(2:4))];
56    % time rate of relative attitude
57    xdot(14:17) = A*wji;
58    
59    % time rate of relative position
60    xdot(11:13) = (vj-vin) - squiggle(wi)*rji;
61
62    xkPlus(:,k) = xhat(:) + Ts*xdot;
63    %re-normalize
64    xkPlus(7:10,k) = xkPlus(7:10,k)/norm(xkPlus(7:10,k));
65    xkPlus(14:17,k) = xkPlus(14:17,k)/norm(xkPlus(14:17,k));
66end
67
68end
